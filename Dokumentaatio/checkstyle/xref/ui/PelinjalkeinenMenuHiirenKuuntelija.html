14  class PelinjalkeinenMenuHiirenKuuntelija implements ActionListener {
15  
16      private JButton uusiPeli;
17      private JButton takaisinMenuun;
18      private JButton suljeOhjelma;
19      private Ui ui;
20      private Pelimoottori moottori;
21  
22      /**
23       * Konstruktorissa alustetaan luokan attribuutit.
24       *
25       * @param uusiPeli
26       * @param takaisinMenuun
27       * @param suljeOhjelma
28       * @param ui
29       */
30      public PelinjalkeinenMenuHiirenKuuntelija(JButton uusiPeli, JButton takaisinMenuun, JButton suljeOhjelma, Ui ui) {
31          this.uusiPeli = uusiPeli;
32          this.takaisinMenuun = takaisinMenuun;
33          this.suljeOhjelma = suljeOhjelma;
34          this.ui = ui;
35          this.moottori = new Pelimoottori(ui, ui.getLogiikka());
36      }
37  
38      /**
39       * Toteutetaan nappien painallusten toiminnallisuudet.
40       *
41       * @param ae
42       */
43      @Override
44      public void actionPerformed(ActionEvent ae) {
45          if (ae.getSource() == uusiPeli) {
46              ui.getVanhaframe().setVisible(false);
47              ui.getFrame().setVisible(false);
48              Ui uusi = new Ui(new Sovelluslogiikka());
49              uusi.kaynnistaEnsimmainenTaso();
50          } else if (ae.getSource() == takaisinMenuun) {
51              ui.getVanhaframe().setVisible(false);
52              ui.getFrame().setVisible(false);
53              Ui uusi = new Ui(new Sovelluslogiikka());
54              uusi.luoMenu();
55          } else if (ae.getSource() == suljeOhjelma) {
56              System.exit(0);
57          }
58      }
59  }
