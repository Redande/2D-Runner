49      /**
50       * Metodissa luodaan menu.
51       */
52      public void luoMenu() {
53          frame = new JFrame("2D-Runner");
54          frame.setPreferredSize(new Dimension(150, 300));
55          frame.setResizable(false);
56          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
57  
58          luoMenunKomponentit(frame.getContentPane());
59  
60          frame.pack();
61          frame.setLocationRelativeTo(null);
62          frame.setVisible(true);
63      }

65      /**
66       * Luodaan menun komponentit.
67       *
68       * @param container
69       */
70      public void luoMenunKomponentit(Container container) {
71          BorderLayout frameLayout = new BorderLayout();
72          container.setLayout(frameLayout);
73  
74          JPanel menuPanel = new JPanel();
75          BoxLayout menuPanelLayout = new BoxLayout(menuPanel, BoxLayout.Y_AXIS);
76          menuPanel.setLayout(menuPanelLayout);
77          menuPanel.add(Box.createRigidArea(new Dimension(25, 25)));
78  
79          JLabel menuLabel = new JLabel("Menu");
80          menuLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
81          menuPanel.add(menuLabel);
82          menuPanel.add(Box.createRigidArea(new Dimension(25, 25)));
83  
84          JButton tasovalikko = new JButton("Tasovalikko");
85          JButton ohjeet = new JButton("Ohjeet");
86          JButton suljeOhjelma = new JButton("Sulje ohjelma");
87          MenuHiirenKuuntelija menukuuntelija = new MenuHiirenKuuntelija(tasovalikko, ohjeet, suljeOhjelma, this);
88          tasovalikko.addActionListener(menukuuntelija);
89          ohjeet.addActionListener(menukuuntelija);
90          suljeOhjelma.addActionListener(menukuuntelija);
91          tasovalikko.setAlignmentX(Component.CENTER_ALIGNMENT);
92          ohjeet.setAlignmentX(Component.CENTER_ALIGNMENT);
93          suljeOhjelma.setAlignmentX(Component.CENTER_ALIGNMENT);
94          menuPanel.add(tasovalikko);
95          menuPanel.add(ohjeet);
96          menuPanel.add(suljeOhjelma);
97  
98          container.add(menuPanel, BorderLayout.CENTER);
99      }

296     /**
297      * Pelimoottoria varten luotu metodi, jossa päivitetään näkymää.
298      */
299     public void piirra() {
300         if (ikkuna == null) {
301             return;
302         }
303         ikkuna.repaint();
304         if (logiikka.getPeliloppui()) {
305             moottori.sammuta();
306             vanhaframe = frame;
307             pelinjalkeinenMenu();
308             return;
309         }
310     }
